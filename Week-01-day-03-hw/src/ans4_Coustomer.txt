public class Coustomer {
    private int id;
    private String name;
    private char gender;

    public Coustomer(int id, String name, char gender) {
        this.id = id;
        this.name = name;
        this.gender = gender;
    }
    public int getId() {
        return id;
    }
    public String getName() {
        return name;
    }
    public char getGender() {
        return gender;
    }
    public String toString() {
        return id + " (" + name + ") at " + gender;
    }
}
###### The Main #######
public class Main {
    public static void main(String[] args) {
        Coustomer C1 = new Coustomer(1, " khaled " , 'm');
        System.out.println(C1.toString());
    }
}
